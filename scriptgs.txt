/********************** MAIN ENTRY POINT - ROUTING **********************/
// URL patterns:
// AOPS/FMOP: ?secret=xxx&ticketId=xxx
// MOS:       ?secret=xxx&action=mos&ticketId=xxx
// MOS Batch: ?secret=xxx&action=mos_batch
// Test:      ?secret=xxx&action=test
function doGet(e) {
  const callback = e.parameter.callback;
  const action = e.parameter.action || '';
  
  // Auth check
  if (!e.parameter.secret || e.parameter.secret !== SECRET_TOKEN) {
    return createResponse({ error: 'Invalid secret token' }, callback);
  }

  try {
    // Route: MOS
    if (action === 'mos') {
      return handleMOSImport(e, callback);
    }
    
    // Route: MOS Batch
    if (action === 'mos_batch') {
      return handleMOSBatchImport(e, callback);
    }
    
    // Route: Test
    if (action === 'test') {
      return handleTest(e, callback);
    }
    
    // Default: AOPS/FMOP Tracker
    return handleAOPSImport(e, callback);
    
  } catch (err) {
    logDebug('doGet: Unhandled error', { action, error: err.message });
    return createResponse({ success: false, error: err.message }, callback);
  }
}

/********************** HANDLER: AOPS/FMOP TRACKER **********************/
function handleAOPSImport(e, callback) {
  const ticketId = e.parameter.ticketId;
  if (!ticketId) {
    return createResponse({ error: 'Missing ticketId' }, callback);
  }

  try {
    logDebug('AOPS/FMOP: Fetching ticket', { ticketId });
    
    // 1. Gọi Supabase (Bên ngoài Lock để không nghẽn hệ thống khi mạng chậm)
    const ticketData = fetchTicketFromView(ticketId);
    if (!ticketData || ticketData.length === 0) {
       throw new Error(`Ticket ${ticketId} không tồn tại trên Supabase View`);
    }

    // 2. Bật Lock (Giới hạn chờ 60 giây để tránh Exceeded Time Limit)
    const lock = LockService.getScriptLock();
    const success = lock.tryLock(60000); 

    if (!success) {
       throw new Error("Server đang bận xử lý request khác. Vui lòng thử lại sau vài giây.");
    }

    let result;
    try {
      // 3. Xử lý ghi dữ liệu vào Sheet
      result = processTicketIntoSheet(ticketData[0]);
    } finally {
      // Nhả Lock ngay lập tức sau khi ghi xong
      lock.releaseLock();
    }

    // 4. Cập nhật trạng thái trên Supabase
    const updateResult = markTicketAsImported(ticketId);
    logDebug('AOPS/FMOP: Ticket imported', { ticketId, updateResult, result });

    return createResponse(result, callback);

  } catch (err) {
    logDebug('AOPS/FMOP: Error', { ticketId, error: err.message });
    return createResponse({ success: false, error: err.message }, callback);
  }
}

/********************** HANDLER: MOS TRACKER **********************/
function handleMOSImport(e, callback) {
  const ticketId = e.parameter.ticketId;
  if (!ticketId) {
    return createResponse({ error: 'Missing ticketId' }, callback);
  }

  try {
    logDebug('MOS: Starting import', { ticketId });
    
    // 1. Gọi Supabase để lấy dữ liệu từ escalate view (Bên ngoài Lock)
    const ticketData = fetchEscalateTicketFromView(ticketId);
    if (!ticketData || ticketData.length === 0) {
       throw new Error(`Ticket ${ticketId} không tồn tại trên Supabase Escalate View hoặc đã được export`);
    }
    
    logDebug('MOS: Ticket data fetched', { ticketId, ticket: ticketData[0]['Ticket Number'] });

    // 2. Bật Lock (Giới hạn chờ 60 giây)
    const lock = LockService.getScriptLock();
    const success = lock.tryLock(60000); 

    if (!success) {
       throw new Error("Server đang bận xử lý request khác. Vui lòng thử lại sau vài giây.");
    }

    let result;
    try {
      // 3. Xử lý ghi dữ liệu vào MOS Sheet
      result = processEscalateTicketIntoMOSSheet(ticketData[0]);
      logDebug('MOS: Sheet write complete', { ticketId, result });
    } finally {
      // Nhả Lock ngay lập tức sau khi ghi xong
      lock.releaseLock();
    }

    // 4. Cập nhật trạng thái trên Supabase - CRITICAL: PHẢI ĐỢI KẾT QUẢ
    const updateResult = markEscalateTicketAsImported(ticketId);
    logDebug('MOS: PATCH result', { ticketId, updateResult });
    
    // Kiểm tra xem PATCH có thực sự thành công không
    if (updateResult && !updateResult.success) {
      throw new Error(`MOS PATCH failed: ${updateResult.error}`);
    }

    return createResponse({
      success: true, 
      imported: true,
      ticketId: ticketId,
      ...result
    }, callback);

  } catch (err) {
    logDebug('MOS: Error', { ticketId, error: err.message, stack: err.stack });
    return createResponse({ success: false, error: err.message }, callback);
  }
}

/********************** HANDLER: MOS BATCH IMPORT **********************/
function handleMOSBatchImport(e, callback) {
  try {
    logDebug('MOS Batch: Starting batch import');
    
    // Lấy danh sách tickets từ escalate view
    const allTickets = fetchAllEscalateTickets();
    if (!allTickets || allTickets.length === 0) {
      return createResponse({ success: true, message: 'Không có ticket nào cần import', imported: 0 }, callback);
    }
    
    logDebug('MOS Batch: Found tickets', { count: allTickets.length });

    const results = [];
    const errors = [];
    
    for (const ticket of allTickets) {
      const ticketId = ticket.id;
      try {
        // Ghi vào Sheet
        const sheetResult = processEscalateTicketIntoMOSSheet(ticket);
        
        // PATCH - CRITICAL
        const updateResult = markEscalateTicketAsImported(ticketId);
        
        if (updateResult && updateResult.success) {
          results.push({
            ticketId,
            ticket: ticket['Ticket Number'],
            status: 'success',
            sheet: sheetResult.sheet,
            row: sheetResult.row
          });
        } else {
          errors.push({
            ticketId,
            ticket: ticket['Ticket Number'],
            error: updateResult ? updateResult.error : 'Unknown PATCH error'
          });
        }
      } catch (err) {
        errors.push({
          ticketId,
          ticket: ticket['Ticket Number'],
          error: err.message
        });
      }
      // Delay nhỏ để tránh rate limit
      Utilities.sleep(500);
    }

    const summary = {
      success: errors.length === 0,
      total: allTickets.length,
      imported: results.length,
      failed: errors.length,
      results: results,
      errors: errors
    };
    
    logDebug('MOS Batch: Complete', summary);
    return createResponse(summary, callback);

  } catch (err) {
    logDebug('MOS Batch: Error', { error: err.message });
    return createResponse({ success: false, error: err.message }, callback);
  }
}

/********************** AOPS/FMOP CORE LOGIC **********************/
function processTicketIntoSheet(ticket) {
  const ticketNumber = String(ticket['Ticket'] || ''); 
  const project = String(ticket['Project'] || '');
  const isOt = ticket['OT Mode'] === true;

  // LOGIC: KIỂM TRA PENDING (ASK IH)
  const rawStatus = String(ticket['Ticket Status'] || '').trim();
  const isPendingAskIH = (rawStatus === 'Pending (Ask IH)');
  
  // Nếu là Pending Ask IH thì xóa status trước khi ghi vào sheet
  const displayStatus = isPendingAskIH ? '' : rawStatus;

  // Chuẩn bị ngày so sánh
  const timeZone = Session.getScriptTimeZone();
  const today = new Date();
  const todayStr = Utilities.formatDate(today, timeZone, DATE_FORMAT_FOR_TRACKER);
  const completedDateStr = Utilities.formatDate(today, timeZone, DATE_COMPLETED_FORMAT);

  // Chọn Sheet phù hợp
  let sheetName = SHEET_AOPS; 
  if (isOt) sheetName = SHEET_OT_TRACKER;
  else if (ticketNumber.toUpperCase().startsWith('POS')) sheetName = SHEET_AOPS;
  else if (project === 'FMOP') sheetName = SHEET_FMOP;

  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) throw new Error(`Không tìm thấy sheet: ${sheetName}`);

  // CACHE HEADERS
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  const headers = getOrCacheHeaders(sheet, lastCol);
  const normalize = (str) => String(str || '').toLowerCase().replace(/\s+/g, '').trim();
  
  const dateColIdx = headers.findIndex(h => normalize(h) === normalize(TRACKER_COLUMNS.DATE));
  const ticketColIdx = headers.findIndex(h => normalize(h) === normalize(TRACKER_COLUMNS.TICKET_NUMBER));

  if (dateColIdx === -1 || ticketColIdx === -1) {
    throw new Error(`Lỗi cấu hình: Không tìm thấy cột Date hoặc Ticket Number`);
  }

  // QUÉT CHỈ 2 CỘT CẦN THIẾT
  const SCAN_LIMIT = 5000;
  let startScanRow = DATA_START_ROW;

  if (lastRow > (DATA_START_ROW + SCAN_LIMIT)) {
    startScanRow = lastRow - SCAN_LIMIT + 1;
  }

  let dateColumn = [];
  let ticketColumn = [];

  if (lastRow >= startScanRow) {
    const numRows = lastRow - startScanRow + 1;
    dateColumn = sheet.getRange(startScanRow, dateColIdx + 1, numRows, 1).getValues().map(r => r[0]);
    ticketColumn = sheet.getRange(startScanRow, ticketColIdx + 1, numRows, 1).getValues().map(r => r[0]);
  }

  // TÌM TRÙNG LẶP & DÒNG TRỐNG
  let targetRow = -1;
  let isUpdate = false;
  let firstEmptyRowIndex = -1;

  for (let i = 0; i < dateColumn.length; i++) {
    const rowTicket = String(ticketColumn[i] || '').trim();
    const rowDateRaw = dateColumn[i];

    let rowDateStr = '';
    if (rowDateRaw instanceof Date) {
      rowDateStr = Utilities.formatDate(rowDateRaw, timeZone, DATE_FORMAT_FOR_TRACKER);
    } else {
      rowDateStr = String(rowDateRaw || '').trim();
    }

    if (rowTicket === ticketNumber && rowDateStr === todayStr) {
      targetRow = startScanRow + i;
      isUpdate = true;
      break;
    }

    if (firstEmptyRowIndex === -1 && rowTicket === '' && rowDateStr === '') {
      firstEmptyRowIndex = startScanRow + i;
    }
  }

  if (targetRow === -1) {
    if (firstEmptyRowIndex !== -1) {
      targetRow = firstEmptyRowIndex;
    } else {
      targetRow = lastRow + 1;
      if (targetRow < DATA_START_ROW) targetRow = DATA_START_ROW;
    }
  }

  // MAPPING DỮ LIỆU
  const dataToMap = new Map([
    [TRACKER_COLUMNS.DATE, todayStr],
    [TRACKER_COLUMNS.TICKET_NUMBER, ticketNumber],
    [TRACKER_COLUMNS.START_TIME, ticket['Start Time'] || ''],
    [TRACKER_COLUMNS.END_TIME, ticket['End Time'] || ''],
    [TRACKER_COLUMNS.NAME, ticket['Name'] || ''],
    [TRACKER_COLUMNS.ACCOUNT, ticket['Account'] || ''],
    [TRACKER_COLUMNS.TICKET_TYPE, ticket['Ticket Type'] || ''],
    [TRACKER_COLUMNS.SUPPLIER, ticket['Supplier'] || ''],
    [TRACKER_COLUMNS.PO, ticket['PO'] || ''],
    [TRACKER_COLUMNS.TICKET_STATUS, displayStatus]
  ]);

  // Bảo toàn công thức
  let currentRowFormulas = [];
  if (targetRow <= sheet.getLastRow()) {
     currentRowFormulas = sheet.getRange(targetRow, 1, 1, lastCol).getFormulas()[0];
  }

  const normalizedHeaders = headers.map(h => normalize(h));
  const dataMapKeys = Array.from(dataToMap.keys()).map(k => normalize(k));

  const newRowData = new Array(lastCol).fill('');
  
  // Xử lý ghi Completed Date
  const dateCompletedIdx = normalizedHeaders.findIndex(h => h.includes('datecompleted') || h.includes('completeddate'));
  if (dateCompletedIdx !== -1) {
    newRowData[dateCompletedIdx] = isPendingAskIH ? '' : completedDateStr;
  }

  for (let i = 0; i < headers.length; i++) {
    const headerNorm = normalizedHeaders[i];
    const mappedKeyIdx = dataMapKeys.indexOf(headerNorm);

    if (mappedKeyIdx !== -1) {
      const originalKey = Array.from(dataToMap.keys())[mappedKeyIdx];
      newRowData[i] = dataToMap.get(originalKey);
    } else if (currentRowFormulas[i] && currentRowFormulas[i] !== '') {
      newRowData[i] = currentRowFormulas[i];
    }
  }

  sheet.getRange(targetRow, 1, 1, newRowData.length).setValues([newRowData]);

  return { 
    success: true, 
    row: targetRow, 
    sheet: sheetName, 
    ticket: ticketNumber, 
    action: isUpdate ? 'updated' : (firstEmptyRowIndex !== -1 ? 'filled_empty' : 'appended')
  };
}

/********************** MOS CORE LOGIC **********************/
function processEscalateTicketIntoMOSSheet(ticket) {
  const ticketNumber = String(ticket['Ticket Number'] || '');

  logDebug('MOS: Opening sheet', { sheetId: SHEET_ID, sheetName: SHEET_MOS });

  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(SHEET_MOS);
  if (!sheet) throw new Error(`Không tìm thấy sheet: ${SHEET_MOS}`);

  // MOS sheet: đọc headers từ row 1, giả định 10 cột
  const MOS_HEADER_ROW = 1;
  const MOS_COLUMNS_COUNT = 10;
  
  const lastRow = sheet.getLastRow();
  const headers = sheet.getRange(MOS_HEADER_ROW, 1, 1, MOS_COLUMNS_COUNT).getValues()[0];
  
  logDebug('MOS: Sheet headers', { headers });
  
  // Filter out empty headers
  const validHeaders = headers.filter(h => String(h || '').trim() !== '');
  logDebug('MOS: Valid headers count', { count: validHeaders.length });

  const normalize = (str) => String(str || '').toLowerCase().replace(/\s+/g, '').trim();
  const ticketColIdx = headers.findIndex(h => normalize(h) === normalize(MOS_COLUMNS.TICKET_NUMBER));

  if (ticketColIdx === -1) {
    throw new Error(`Lỗi cấu hình: Không tìm thấy cột Ticket Number trong MOS sheet. Headers: ${headers.join(', ')}`);
  }
  
  const lastCol = headers.length;

  // QUÉT CỘT TICKET ĐỂ TÌM TRÙNG LẶP
  const SCAN_LIMIT = 5000;
  let startScanRow = DATA_START_ROW;

  if (lastRow > (DATA_START_ROW + SCAN_LIMIT)) {
    startScanRow = lastRow - SCAN_LIMIT + 1;
  }

  let ticketColumn = [];
  if (lastRow >= startScanRow) {
    const numRows = lastRow - startScanRow + 1;
    ticketColumn = sheet.getRange(startScanRow, ticketColIdx + 1, numRows, 1).getValues().map(r => r[0]);
  }

  // TÌM TRÙNG LẶP & DÒNG TRỐNG
  let targetRow = -1;
  let firstEmptyRowIndex = -1;

  for (let i = 0; i < ticketColumn.length; i++) {
    const rowTicket = String(ticketColumn[i] || '').trim();

    if (rowTicket === ticketNumber) {
      targetRow = startScanRow + i;
      break;
    }

    if (firstEmptyRowIndex === -1 && rowTicket === '') {
      firstEmptyRowIndex = startScanRow + i;
    }
  }

  if (targetRow === -1) {
    if (firstEmptyRowIndex !== -1) {
      targetRow = firstEmptyRowIndex;
    } else {
      targetRow = lastRow + 1;
      if (targetRow < DATA_START_ROW) targetRow = DATA_START_ROW;
    }
  }

  logDebug('MOS: Target row determined', { ticketNumber, targetRow, isUpdate: targetRow <= lastRow });

  // MAPPING DỮ LIỆU
  const dataToMap = new Map([
    [MOS_COLUMNS.DATE_COMPLETE, ticket['Date Complete'] || ''],
    [MOS_COLUMNS.AGENT_VCN, ticket['Agent VCN'] || ''],
    [MOS_COLUMNS.ACCOUNT, ticket['Account'] || ''],
    [MOS_COLUMNS.TICKET_NUMBER, ticketNumber],
    [MOS_COLUMNS.TICKET_URL, ticket['Ticket URL'] || ''],
    [MOS_COLUMNS.TICKET_TYPE, ticket['Ticket Type'] || ''],
    [MOS_COLUMNS.MOS_DETAIL, ticket['MOS Detail'] || '']
  ]);

  // Bảo toàn công thức
  let currentRowFormulas = [];
  if (targetRow <= sheet.getLastRow()) {
     currentRowFormulas = sheet.getRange(targetRow, 1, 1, lastCol).getFormulas()[0];
  }

  const normalizedHeaders = headers.map(h => normalize(h));
  const dataMapKeys = Array.from(dataToMap.keys()).map(k => normalize(k));

  const newRowData = new Array(lastCol).fill('');

  for (let i = 0; i < headers.length; i++) {
    const headerNorm = normalizedHeaders[i];
    const mappedKeyIdx = dataMapKeys.indexOf(headerNorm);

    if (mappedKeyIdx !== -1) {
      const originalKey = Array.from(dataToMap.keys())[mappedKeyIdx];
      newRowData[i] = dataToMap.get(originalKey);
    } else if (currentRowFormulas[i] && currentRowFormulas[i] !== '') {
      newRowData[i] = currentRowFormulas[i];
    }
  }

  // Ghi dữ liệu
  sheet.getRange(targetRow, 1, 1, newRowData.length).setValues([newRowData]);
  
  logDebug('MOS: Data written to sheet', { ticketNumber, row: targetRow });

  return { 
    success: true, 
    row: targetRow, 
    sheet: SHEET_MOS, 
    ticket: ticketNumber, 
    action: targetRow <= lastRow ? 'updated' : 'appended'
  };
}

/********************** HELPERS **********************/
// Cache headers để tránh đọc lại nhiều lần
const HEADER_CACHE = {};
function getOrCacheHeaders(sheet, lastCol) {
  const sheetName = sheet.getName();
  if (!HEADER_CACHE[sheetName]) {
    HEADER_CACHE[sheetName] = sheet.getRange(HEADER_ROW, 1, 1, lastCol).getValues()[0];
  }
  return HEADER_CACHE[sheetName];
}

// Xóa cache headers (gọi khi cần refresh)
function clearHeaderCache() {
  for (const key in HEADER_CACHE) {
    delete HEADER_CACHE[key];
  }
  logDebug('Header cache cleared');
}

function createResponse(data, callbackName) {
  const jsonString = JSON.stringify(data);
  return ContentService.createTextOutput(callbackName ? `${callbackName}(${jsonString})` : jsonString)
    .setMimeType(callbackName ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON);
}

/********************** SUPABASE FUNCTIONS - AOPS/FMOP **********************/
function fetchTicketFromView(ticketId) {
  const url = `${SUPABASE_URL}/rest/v1/tickets_export_v?id=eq.${ticketId}&select=*`;
  const options = {
    method: 'GET',
    headers: {
      'apikey': SUPABASE_KEY,
      'Authorization': 'Bearer ' + SUPABASE_KEY,
      'Content-Type': 'application/json'
    },
    muteHttpExceptions: true,
    validateHttpsCertificates: false
  };

  const response = UrlFetchApp.fetch(url, options);
  logDebug('Supabase fetch ticket', { ticketId, status: response.getResponseCode() });

  if (response.getResponseCode() !== 200) {
     throw new Error("Lỗi Supabase: " + response.getContentText());
  }
  return JSON.parse(response.getContentText());
}

function markTicketAsImported(ticketId) {
  try {
    const url = `${SUPABASE_URL}/rest/v1/tickets?id=eq.${ticketId}`;
    const options = {
      method: 'PATCH',
      headers: {
        'apikey': SUPABASE_KEY,
        'Authorization': 'Bearer ' + SUPABASE_KEY,
        'Content-Type': 'application/json',
        'Prefer': 'return=minimal'
      },
      payload: JSON.stringify({ import_to_tracker: true }),
      muteHttpExceptions: false, // CRITICAL: Phải throw exception khi lỗi
      validateHttpsCertificates: false
    };

    const response = UrlFetchApp.fetch(url, options);
    const statusCode = response.getResponseCode();
    
    logDebug('PATCH AOPS/FMOP', { ticketId, statusCode });
    
    return { success: statusCode === 204 || statusCode === 200, statusCode };
    
  } catch (e) {
    logDebug('PATCH AOPS/FMOP Error', { ticketId, error: e.message });
    return { success: false, error: e.message };
  }
}

    validateHttpsCertificates: false
  };

  const response = UrlFetchApp.fetch(url, options);
  logDebug('Supabase fetch escalate', { ticketId, status: response.getResponseCode() });

  if (response.getResponseCode() !== 200) {
     throw new Error("Lỗi Supabase escalate view: " + response.getContentText());
  }
  return JSON.parse(response.getContentText());
}

function markEscalateTicketAsImported(ticketId) {
  try {
    const url = `${SUPABASE_URL}/rest/v1/tickets?id=eq.${ticketId}`;
    const options = {
      method: 'PATCH',
      headers: {
        'apikey': SUPABASE_KEY,
        'Authorization': 'Bearer ' + SUPABASE_KEY,
        'Content-Type': 'application/json',
        'Prefer': 'return=minimal'
      },
      payload: JSON.stringify({ import_to_mos_tracker: true }),
      muteHttpExceptions: false, // CRITICAL: Phải throw exception khi lỗi
      validateHttpsCertificates: false
    };

    logDebug('MOS PATCH: Starting', { ticketId, url });

    const response = UrlFetchApp.fetch(url, options);
    const statusCode = response.getResponseCode();
    const responseText = response.getContentText();
    
    logDebug('MOS PATCH: Response', { ticketId, statusCode, responseText });
    
    // Supabase Rest API trả về 204 No Content khi PATCH thành công
    if (statusCode === 204 || statusCode === 200) {
      return { success: true, statusCode };
    } else {
      return { success: false, statusCode, error: responseText };
    }
    
  } catch (e) {
    logDebug('MOS PATCH: Error', { ticketId, error: e.message, stack: e.stack });
    return { success: false, error: e.message };
  }
}

// Chạy function này để test batch import tất cả tickets vào MOS
function testMOSImportBatch() {
  clearHeaderCache();
  
  const tickets = fetchAllEscalateTickets();
  Logger.log('Found ' + tickets.length + ' tickets to import');
  
  const results = [];
  
  for (const ticket of tickets) {
    try {
      const sheetResult = processEscalateTicketIntoMOSSheet(ticket);
      const patchResult = markEscalateTicketAsImported(ticket.id);
      
      results.push({
        id: ticket.id,
        ticket: ticket['Ticket Number'],
        sheetSuccess: sheetResult.success,
        patchSuccess: patchResult.success,
        error: patchResult.error || null
      });
      
      Logger.log(`Ticket ${ticket['Ticket Number']}: Sheet=${sheetResult.success}, PATCH=${patchResult.success}`);
      
    } catch (e) {
      Logger.log(`Error for ${ticket['Ticket Number']}: ${e.message}`);
      results.push({
        id: ticket.id,
        ticket: ticket['Ticket Number'],
        sheetSuccess: false,
        patchSuccess: false,
        error: e.message
      });
    }
    
    Utilities.sleep(500);
  }
  
  Logger.log('=== BATCH IMPORT RESULTS ===');
  Logger.log(JSON.stringify(results, null, 2));
  
  // Summary
  const successCount = results.filter(r => r.patchSuccess).length;
  Logger.log(`Success: ${successCount}/${tickets.length}`);
}
